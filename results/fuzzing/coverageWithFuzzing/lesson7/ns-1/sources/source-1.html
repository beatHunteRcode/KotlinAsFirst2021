25 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
26     val writer = File(outputName).bufferedWriter()
27     var currentLineLength = 0
28     fun append(word: String) {
29         if (currentLineLength > 0) {
30             if (word.length + currentLineLength >= lineLength) {
31                 writer.newLine()
32                 currentLineLength = 0
33             } else {
34                 writer.write(" ")
35                 currentLineLength++
36             }
37         }
38         writer.write(word)
39         currentLineLength += word.length
40     }
41     for (line in File(inputName).readLines()) {
42         if (line.isEmpty()) {
43             writer.newLine()
44             if (currentLineLength > 0) {
45                 writer.newLine()
46                 currentLineLength = 0
47             }
48             continue
49         }
50         for (word in line.split(Regex("\\s+"))) {
51             append(word)
52         }
53     }
54     writer.close()
55 }

302 fun markdownToHtmlSimple(inputName: String, outputName: String) {
303     val startText = File(inputName).readText()
304     val toReturn = File(outputName).bufferedWriter()
305 
306     val replaceBold = Regex("\\*\\*([\\s\\S]*?)\\*\\*").replace(startText) { char ->
307         "<b>" + char.value.replace("**", "") + "</b>"
308     }
309     val replaceItalics = Regex("\\*([\\s\\S]*?)\\*").replace(replaceBold) { char ->
310         "<i>" + char.value.replace("*", "") + "</i>"
311     }
312     val replaceStrikethrough = Regex("~~([\\s\\S]*?)~~").replace(replaceItalics) { char ->
313         "<s>" + char.value.replace("~~", "") + "</s>"
314     }
315 
316     val textWithTags = replaceStrikethrough
317     var linesCheckedInParagraph = 0
318     val lines = textWithTags.split("\n").toMutableList()
319     for (idx in lines.indices) {
320         val singleLine = lines[idx]
321         if (singleLine.trim().isEmpty()) {
322             if (linesCheckedInParagraph >= 1) {
323                 if (idx + 1 < lines.size && lines[idx + 1].trim().isNotEmpty()) {
324                     lines[idx] = "</p><p>"
325                     linesCheckedInParagraph = 0
326                 }
327             }
328         } else {
329             linesCheckedInParagraph += 1
330         }
331     }
332 
333     toReturn.write("<html><body><p>${lines.joinToString(separator = "")}</p></body></html>")
334     toReturn.close()
335 }

433  * (Отступы и переносы строк в примере добавлены для наглядности, при решении задачи их реализовывать не обязательно)
434  */
435 fun markdownToHtmlLists(inputName: String, outputName: String) {
436     val lines = File(inputName).readLines()
437     val toReturn = File(outputName).bufferedWriter()
438     val builder = StringBuilder("<html><body><p>")
439 
440     val digits = ".0123456789"
441     val currentStack = mutableListOf<String>()
442 
443 // Для первой строчки
444     var idx = 0
445     if (lines.isNotEmpty()) {
446         if (lines[0][idx] == '*') {
447             currentStack.add("ul")
448         } else currentStack.add("ol")
449 
450         while (lines[0][idx] == ' ' || lines[0][idx] == '*' || lines[0][idx] in digits) idx += 1
451         builder.append("<${currentStack.last()}><li>${lines[0].substring(idx)}")
452 
453 // для последующих строчек
454         for (j in 1 until lines.size) {
455             val currentLine = lines[j]
456             if (currentLine.trim().isEmpty()) continue
457             var c = 0
458             while (currentLine[c] == ' ') c += 1
459             if (c > 4 * (currentStack.size - 1)) {
460                 if (currentLine[c] in digits) {
461                     builder.append("<ol>")
462                     currentStack.add("ol")
463                 } else if (currentLine[c] == '*') {
464                     builder.append("<ul>")
465                     currentStack.add("ul")
466                 } else throw IllegalArgumentException()
467             } else if (c < 4 * (currentStack.size - 1)) {
468                 builder.append("</li></${currentStack.removeLast()}></li>")
469             } else {
470                 builder.append("</li>")
471             }
472 
473             while (currentLine[c] == ' ' || currentLine[c] == '*' || currentLine[c] == '.' || currentLine[c] in digits) c += 1
474             builder.append("<li>${currentLine.substring(c)}")
475         }
476 
477         while (currentStack.size > 0) builder.append("</li></${currentStack.removeLast()}>")
478     }
479 
480     builder.append("</p></body></html>")
481     toReturn.write("$builder")
482     toReturn.close()
483 }
