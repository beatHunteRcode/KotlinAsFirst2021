157 fun plusMinus(expression: String): Int {
158     val parts = expression.split(" ")
159     if (parts[0].contains('+') || parts[0].contains('-')) throw IllegalArgumentException()
160 
161     var sum = parts[0].toInt()
162     for (i in 1 until parts.size step 2) {
163         if ((parts[i].contains('+') || parts[i].contains('-'))
164             && (parts[i + 1].contains('+') || parts[i + 1].contains('-'))
165             || isNumber(parts[i])
166         )
167             throw IllegalArgumentException()
168         else when {
169             parts[i].contains('+') -> sum += parts[i + 1].toInt()
170             parts[i].contains('-') -> sum -= parts[i + 1].toInt()
171         }
172     }
173     return sum
174 }
